```
class Solution {
public:
    bool isValid(string s) {
        if(s.length() == 1)
            return false;
        std::map<char, char> m;
        std::stack<char> stack1;
        char c1[] = {'{', '}','[', ']', '(',')'};
        char c2[] = {'0','{','0','[','0', '('};
        for(int i = 0; i < 6; i ++){
            m.insert(pair<char, char> (c1[i],c2[i]));    
        }
        
        for(int i = 0; i < s.length(); i++){
            if(m[s[i]] == '0')
                stack1.push(s[i]);
            else if(stack1.empty() || m[s[i]] != stack1.top() )
                return false;
            else if(m[s[i]] == stack1.top())
                stack1.pop();
        }
        
        if(stack1.empty())
            return true;
        return false;
    }
};
```
